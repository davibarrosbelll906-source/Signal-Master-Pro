2. Rota no Backend (Express)
Crie ou edite: artifacts/api-server/src/routes/luna.ts
TypeScriptimport express from 'express';
import { lunaAgent } from '../lib/lunaAgent';

const router = express.Router();

router.post('/chat', async (req, res) => {
  const { message, imageBase64, pair, timeframe } = req.body; // imageBase64 opcional

  try {
    let userContent: any = [{ type: "text", text: message }];

    if (imageBase64) {
      userContent.push({
        type: "image_url",
        image_url: { url: imageBase64 }   // base64: data:image/png;base64,...
      });
    }

    const stream = await lunaAgent.stream({
      messages: [{ role: "user", content: userContent }]
    });

    // Streaming para o usuário ver a resposta aos poucos
    for await (const chunk of stream) {
      if (chunk.content) res.write(chunk.content);
    }
    res.end();
  } catch (error) {
    console.error(error);
    res.status(500).json({ error: "Erro ao falar com Luna" });
  }
});

export default router;
Não esqueça de registrar a rota no seu server.ts ou app.ts do backend:

3. Onde colocar no aplicativo (Frontend)
Melhor lugar recomendado:

Um botão flutuante fixo no canto inferior direito (em todas as páginas principais, especialmente em SignalsPage).
Dentro do chat com Luna (melhor experiência).

Componente recomendado: LunaChat.tsx
Crie este componente e coloque ele como um modal ou drawer que abre ao clicar no botão flutuante.
Dentro do chat, adicione dois botões principais:

“💬 Falar com Luna” (chat normal)
“📸 Analisar Gráfico com Luna” (captura automática + chat)

Aqui está o código principal para captura (coloque dentro do chat ou em SignalsPage):
tsx// components/LunaChartAnalyzer.tsx
import html2canvas from 'html2canvas';
import { useRef, useState } from 'react';

export default function LunaChartAnalyzer({ chartContainerRef }: { chartContainerRef: React.RefObject<HTMLDivElement> }) {
  const [analyzing, setAnalyzing] = useState(false);

  const captureAndAnalyze = async () => {
    if (!chartContainerRef.current) {
      alert("Gráfico não encontrado");
      return;
    }

    setAnalyzing(true);

    try {
      const canvas = await html2canvas(chartContainerRef.current, {
        scale: 2,                    // Alta qualidade
        useCORS: true,
        backgroundColor: '#111827',
        logging: false,
      });

      const base64Image = canvas.toDataURL('image/png');

      // Envia para Luna
      const response = await fetch('/api/luna/chat', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          message: "Analise este gráfico de forma educativa. Explique tendência, suportes, resistências, padrões de velas e o que eu posso aprender.",
          imageBase64: base64Image,
          pair: "EURUSD",           // passe o par atual do estado
          timeframe: "M5"
        })
      });

      const reply = await response.text();
      // Adicione "reply" ao histórico do chat com Luna
      console.log("Resposta da Luna:", reply);
    } catch (err) {
      alert("Erro ao capturar ou analisar o gráfico");
    } finally {
      setAnalyzing(false);
    }
  };

  return (
    <button
      onClick={captureAndAnalyze}
      disabled={analyzing}
      className="flex items-center gap-2 bg-violet-600 hover:bg-violet-700 text-white px-5 py-3 rounded-xl shadow-lg fixed bottom-6 right-6 z-50"
    >
      {analyzing ? "📸 Analisando com Luna..." : "📸 Analisar Gráfico com Luna"}
    </button>
  );
}
