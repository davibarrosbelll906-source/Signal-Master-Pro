TypeScript// ====================== SIGNAL ENGINE - VERSÃO CORRIGIDA ======================
private currentTimeframe: string = 'M1';   // será sincronizado pelo frontend

private async generateSignals() {
  const now = new Date();
  const currentSecond = now.getSeconds();
  const currentMinute = now.getMinutes();

  // Só executa no segundo :48 (como você já fazia)
  if (currentSecond !== 48) return;

  // Verifica se deve disparar sinal baseado no timeframe atual
  if (!this.shouldTriggerSignal(currentMinute, this.currentTimeframe)) {
    return;
  }

  console.log(`🟢 [SIGNAL ENGINE] Disparando sinal - Timeframe: ${this.currentTimeframe} | Minuto: ${currentMinute}`);

  const activePairs = ['EURUSD', 'GBPUSD', 'USDJPY', 'BTCUSDT', 'ETHUSDT' /* seus pares */];

  for (const pair of activePairs) {
    const candles = this.buffers.get(pair) || [];
    if (candles.length < 60) continue;

    const analysis = this.runFullQuantitativeAnalysis(candles, pair, this.currentTimeframe);

    if (analysis.finalScore < 68) continue;
    if (!this.isValidConfluence(analysis)) continue;
    if (this.isTooFrequent(pair)) continue;

    const signal = {
      id: Date.now().toString(),
      pair,
      direction: analysis.direction,
      score: Math.round(analysis.finalScore),
      tier: this.getTier(analysis.finalScore),
      timeframe: this.currentTimeframe,
      expiry: this.suggestExpiry(this.currentTimeframe),
      reason: analysis.mainReason,
      session: this.getCurrentSession(now),
      confidence: analysis.confidence,
      timestamp: now.toISOString(),
    };

    // Envia sinal para todos os clientes conectados
    this.io.emit('new_signal', signal);

    // Explicação da Luna (opcional)
    this.sendLunaEducationalExplanation(signal, analysis);

    this.lastSignalTime.set(pair, Date.now());
  }
}

// ====================== MÉTODOS AUXILIARES CORRIGIDOS ======================

/**
 * Verifica se deve disparar o sinal no minuto atual baseado no timeframe
 */
private shouldTriggerSignal(currentMinute: number, timeframe: string): boolean {
  switch (timeframe) {
    case 'M1':
      return true;                    // todo minuto

    case 'M5':
      return currentMinute % 5 === 0; // 00, 05, 10, 15, 20, 25, 30...

    case 'M15':
      return currentMinute % 15 === 0; // 00, 15, 30, 45

    default:
      return true;
  }
}

/**
 * Sugere tempo de expiração baseado no timeframe
 */
private suggestExpiry(timeframe: string): number {
  switch (timeframe) {
    case 'M1': return 3;
    case 'M5': return 5;
    case 'M15': return 10;
    default: return 5;
  }
}

/**
 * Atualiza o timeframe atual (chamado pelo frontend via Socket.io)
 */
public updateTimeframe(newTimeframe: string): void {
  if (['M1', 'M5', 'M15'].includes(newTimeframe)) {
    this.currentTimeframe = newTimeframe;
    console.log(`⏱️ Timeframe atualizado para: ${newTimeframe}`);
  }
}

/**
 * Retorna o timeframe atual
 */
public getCurrentTimeframe(): string {
  return this.currentTimeframe;
}
