export class SignalEngine {
  private io: Server;
  private buffers: Map<string, any[]> = new Map();
  private lastSignalTime: Map<string, number> = new Map();

  constructor(io: Server) {
    this.io = io;
    this.startEngine();
  }

  private startEngine() {
    setInterval(() => this.generateSignals(), 1000);
  }

  private async generateSignals() {
    const now = new Date();
    if (now.getSeconds() !== 48) return;   // mantém seu timing original

    const activePairs = ['EURUSD', 'GBPUSD', 'USDJPY', 'BTCUSDT', 'ETHUSDT' /* adicione os seus */];

    for (const pair of activePairs) {
      const candles = this.buffers.get(pair) || [];
      if (candles.length < 60) continue;

      const timeframe = this.getTimeframe(now.getMinutes());

      // === 1. Score clássico + filtros rápidos (rápido e barato) ===
      let score = this.calculateBaseScore(candles, pair, timeframe);

      // Filtros rápidos para aumentar assertividade (sem IA)
      if (!this.passesQuickFilters(candles, score, pair)) continue;

      const rawSignal = {
        id: Date.now().toString(),
        pair,
        direction: score > 75 ? 'CALL' : 'PUT',
        score: Math.round(score),
        tier: this.getTier(score),
        timeframe,
        expiry: timeframe === 'M1' ? 3 : timeframe === 'M5' ? 5 : 10,
        timestamp: now.toISOString(),
      };

      // === 2. Luna Numeric Shadow Validation (corrigido) ===
      const validation = await this.validateWithLunaNumeric(rawSignal, candles);

      if (!validation.approved || validation.confidence < 72) {
        continue; // sinal descartado silenciosamente
      }

      // Sinal aprovado pela Luna
      const finalSignal = {
        ...rawSignal,
        lunaConfidence: validation.confidence,
        lunaReason: validation.shortReason,
        validatedByLuna: true
      };

      this.io.emit('new_signal', finalSignal);
      this.lastSignalTime.set(pair, Date.now());
    }
  }

  // ====================== LUNA NUMERIC SHADOW ======================
  private async validateWithLunaNumeric(signal: any, candles: any[]) {
    // Prepara apenas dados numéricos (últimos 50 candles)
    const recentCandles = candles.slice(-50).map(c => ({
      time: c.time,
      open: c.open,
      high: c.high,
      low: c.low,
      close: c.close,
    }));

    try {
      const result = await lunaAgent.invoke({
        messages: [{
          role: "user",
          content: `
            Você é Luna Shadow Validator.
            Analise numericamente se este sinal tem alta probabilidade de acerto.

            Par: ${signal.pair}
            Direção proposta: ${signal.direction}
            Score inicial: ${signal.score}
            Timeframe: ${signal.timeframe}

            Dados dos últimos 50 candles:
            ${JSON.stringify(recentCandles)}

            Responda APENAS em JSON válido:
            {
              "approved": true/false,
              "confidence": número entre 0 e 100,
              "shortReason": "explicação curta em português"
            }
            Seja rigorosa. Só aprove se houver confluência clara.
          `
        }]
      });

      return JSON.parse(result.content);
    } catch (e) {
      console.error('Luna Shadow error', e);
      return { approved: false, confidence: 0, shortReason: 'Erro na validação' };
    }
  }
